	static void sort(int[] a, int beg, int end) {

        if (beg < end) {
            int x = a[(beg + end) / 2];
            int i = beg;
            int j = end;

            while (i <= j) {
                while (a[i] < x) i++;
                while (a[j] > x) j--;
                if (i <= j) {
                    int t = a[i];
                    a[i] = a[j];
                    a[j] = t;
                    i++;
                    j--;

                }
            }
            sort(a, beg, j);
            sort(a, i, end);
        }
    }

    private void sort(SortData a, int beg, int end) {

        if (beg < end) {
            int x = (beg + end) / 2;
            int i = beg, j = end;
            while (i <= j) {
                while (a.less(i, x)) i++;
                while (a.less(x, j)) j--;
                if (i <= j) {
                    a.swap(i, j);
                    if (i == x) {
                        x = j;
                    } else if (j == x) {
                        x = i;
                    }

                    ++i;
                    --j;
                }
            }
            sort(a, beg, j);
            sort(a, i, end);
        }
    }

            if(a[k] <= p1){

            }else if(a[k] <= p2){
                k++;
            }else{
                j--; swap(a,k,j);
            }
        }
        sort(a, beg, i-1);
        sort(a, i, j-1);
        sort(a, j, end);
    }

private void quicksort(SortData data, int l, int h) { 
	if (h - l > CUT_OFF) {
		int i = l, j = h, m = (l + h) / 2; 
		while (i <= j) {
			while (data.less(i, m)) i++; 
			while (data.less(m, j)) j--; 
			if (i <= j) {
				data.swap(i, j);
			if (i == m) m = j;
				else if (j == m) m = i; i++; j--;
			}
		} 
		quicksort(data, l, j);
		quicksort(data, i, h);
	} else
		insertionSort(data, l, h);
	}
}
private void insertionSort(SortData data, int beg, int end) {
	for (int i = beg + 1; i <= end; i++) {
		int j = i;
		while (j > beg && data.less(j, j - 1)) { data.swap(j, j - 1); j--; } }
}
